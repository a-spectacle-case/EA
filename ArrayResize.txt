double a[]; //double型の動的配列aを宣言
ArrayResize(a, 3); //aの配列サイズを3にする
//値を格納
a[0] = 100.0;
a[1] = 80.0;
a[2] = 120.0;
Print(a[0], " : ", a[1], " : ", a[2]); // 100.0 : 80.0 : 120.0
ArrayResize(a, 2); //aの配列サイズを2にする ＝ aの末尾を削除
Print(a[0], " : ", a[1]); // 100.0 : 80.0
